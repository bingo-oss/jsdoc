});

     organizationSelected: [],  //默认选中的部门id
     userIgnore: [],            //禁止选择的用户id
     groupIgnore: [],           //禁止选择的群组id
     organizationIgnore: [],    //禁止选择的部门id
     isIncludeDisableUser:false //是否包含被禁用的人员， 默认是false
